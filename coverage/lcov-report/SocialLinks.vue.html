<template>
  <section class="social-links">
    <p class="social-links__title">{{ title }}:</p>
    <ul class="list list--inline list--animated">
      <li v-for="(link, index) in links" v-bind:key="`social-icon-${index}`">
        <a v-bind:href="`${link.url}`" v-bind:title="link.text">
          <img
            v-bind:src="`${link.image}`"
            v-bind:alt="`${link.text} icon`"
            class="custom-icon"
            loading="lazy"
          />
          <div class="sr-only">{{ link.text }}</div>
        </a>
      </li>
    </ul>
  </section>
</template>
 
<script>
import fb from '../assets/svg/social-facebook.svg'
import twitter from '../assets/svg/social-twitter.svg'
import linkedin from '../assets/svg/social-linkedin.svg'
import insta from '../assets/svg/social-instagram.svg'
import ytb from '../assets/svg/social-youtube.svg'
 
export default {
  name: 'social-links',
  props: {
    title: {
      type: String,
      required: false,
      default: 'Follow us on',
    },
    links: {
      type: Array,
      required: false,
      default: () => [
        {
          text: 'facebook',
          image: fb,
          url: 'https://www.facebook.com/boughtbymany/',
        },
        {
          text: 'twitter',
          image: twitter,
          url: 'https://twitter.com/boughtbymany',
        },
        {
          text: 'linkedin',
          image: linkedin,
          url: 'https://www.linkedin.com/company/bought-by-many/',
        },
        {
          text: 'instagram',
          image: insta,
          url: 'https://www.instagram.com/boughtbymany/',
        },
        {
          text: 'youtube',
          image: ytb,
          url: 'https://www.youtube.com/channel/UC7QDeIM07gJWeP5yGLPj7fQ',
        },
      ],
      validator: arr => arr.every(icon => icon.text && icon.image && icon.url),
    },
  },
}
</script>
