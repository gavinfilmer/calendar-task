<template>
  <div class="prog-card-under" v-bind:class="{ shadow }">
    <div class="prog-card" v-bind:class="[chevron, { grey }]">
      <section class="content-container">
        <div class="arrow-container">
          <div
            v-if="this.$slots.detail"
            class="arrow-down"
            v-bind:class="{ 'arrow-up': cardOpen }"
            v-on:click="openCard()"
          />
        </div>
        <div class="prog-card__top">
          <tag
            v-for="(tag, index) in tags"
            v-bind:key="index"
            v-bind:tagStyle="tag.color"
            ><slot v-bind:name="tag.name" />
          </tag>
        </div>
        <slot />
        <div v-if="footer" class="footer font--micro">{{ footer }}</div>
      </section>
      <collapse-transition>
        <section v-if="cardOpen" class="card-detail">
          <slot name="detail" />
        </section>
      </collapse-transition>
    </div>
  </div>
</template>
 
<script>
import Tag from './Tag.vue'
import { CollapseTransition } from 'vue2-transitions'
export default {
  name: 'card',
  components: {
    Tag,
    CollapseTransition,
  },
  props: {
    tags: {
      // Array of objects with name/colour keys
      type: Array,
      default() {
        return []
      },
      // Check every tag has a text value
      validator: tags => tags.every(tag => Object.keys(tag).includes('text')),
    },
    shadow: {
      type: Boolean,
      default: true,
    },
    grey: {
      type: Boolean,
      default: false,
    },
    chevron: {
      type: String,
      default: '',
      validator: value => ['', 'chevron-left', 'chevron-right'].includes(value),
    },
    footer: {
      type: String,
      default: '',
    },
    open: {
      type: Boolean,
      default: null,
    },
  },
  data() {
    return {
      localOpen: false,
    }
  },
  computed: {
    cardOpen() {
      // If prop hasn't been set use local state
      return this.open === null ? this.localOpen : this.open
    },
  },
  methods: {
    openCard() {
      // Must maintain own state as well as talking to parent.
      this.localOpen = !this.localOpen
      this.$emit('requestOpen', this.open ? false : true)
    },
  },
}
</script>
