<script>
import { cloneVNode } from '../lib/Utils'
 
export default {
  name: 'MAccordion',
  data() {
    return {
      openIndex: null,
    }
  },
  methods: {
    setOpenIndex(index, shouldOpen = true) {
      // Set the index passed as the open index if `shouldOpen` is `true`. If
      // it's false, we "close" it, by setting the open index to null, but only
      // if it's the currently selected index.
      if (shouldOpen) {
        this.openIndex = index
      } else if (this.openIndex === index) {
        this.openIndex = null
      }
    },
  },
  render(h) {
    // We use a render function here rather than the standard template because
    // we need to iterate over our default slot's children and add a prop and a
    // listener to each node in there. This can't be done in a template.
 
    const moreInfoSection = this.$slots.moreInfo
      ? h('div', { attrs: { class: 'more-info' } }, this.$slots.moreInfo)
      : null
 
    return h(
      'div',
      {
        attrs: {
          class: 'm-accordion-wrapper',
        },
      },
      [
        ...this.$slots.default.map((vNode, index) => {
          // Clone the vNode from the slot and then add our own prop/listener on
          // top.
          const newVNode = cloneVNode(vNode)
          const nodeOptions = newVNode.componentOptions
 
          // Only add our data if the node is a component rather than an HTML
          // element.
          if (nodeOptions) {
            nodeOptions.propsData = nodeOptions.propsData || {}
            nodeOptions.listeners = nodeOptions.listeners || {}
 
            nodeOptions.propsData.open = index === this.openIndex
            nodeOptions.listeners.requestOpen = shouldOpen =>
              this.setOpenIndex(index, shouldOpen)
          }
          return newVNode
        }),
        moreInfoSection,
      ]
    )
  },
}
</script>
