<script>
import Burger from './Burger.vue'
import CloseButton from './CloseButton.vue'
import { telephoneLinkFormatter } from '@boughtbymany/verktyg'
 
export default {
  name: 'SiteHeader',
  components: {
    Burger,
    CloseButton,
  },
  props: {
    user: {
      type: Object,
      required: false,
      default: null,
    },
    overlay: {
      type: Boolean,
    },
    overlayFade: {
      type: Boolean,
      default: true,
    },
    mode: {
      type: String,
      default: 'full',
      validator: value => ['full', 'simple'].includes(value),
    },
    telephoneNumber: {
      type: String,
      default: '0345 340 4090',
    },
  },
  data() {
    return {
      loading: true,
      session: null,
      mobileNavDisplay: false,
      overlayFadeStyle: '',
      overlayFadeComplete: false,
      displayDropdown: false,
    }
  },
  watch: {
    mobileNavDisplay: {
      handler(newValue) {
        const bodyClasses = document.querySelector('body').classList
        if (newValue) {
          bodyClasses.add('drawer-open')
        } else {
          bodyClasses.remove('drawer-open')
        }
      },
    },
  },
  mounted() {
    // Allow for a user to be passed, otherwise attempt to
    // sniff the cookies for a clue to signed in status
    if (!this.user) {
      this.session = this.getSession()
    } else {
      this.session = this.user
    }
 
    if (this.overlay && this.overlayFade) {
      this.triggerOverlay()
      window.onscroll = this.triggerOverlay
    }
 
    this.loading = false
  },
  methods: {
    hideMobileNav() {
      this.mobileNavDisplay = false
    },
    handleAccountBtn() {
      if (this.session) {
        this.toggleDropdown()
        return
      }
      window.location.assign('/account/')
    },
    toggleDropdown() {
      this.displayDropdown = !this.displayDropdown
    },
    triggerOverlay() {
      // Adapted from:
      // https://developer.mozilla.org/en-US/docs/Web/API/Window/scrollY
      const supportPageOffset = window.pageXOffset !== undefined
      const isCSS1Compat = (document.compatMode || '') === 'CSS1Compat'
      const y = supportPageOffset
        ? window.pageYOffset
        : isCSS1Compat
        ? document.documentElement.scrollTop
        : document.body.scrollTop
 
      const revealPoint = 160
      const level = (y * 100) / revealPoint / 100
 
      if (level > 0.75) {
        this.overlayFadeComplete = true
      } else {
        this.overlayFadeComplete = false
      }
 
      this.overlayFadeStyle = `background-color:rgba(255,255,255,${level});`
    },
    /**
     * Check if there is a live session.
     *
     * This does not return a user - just confirms if a session is present. If
     * can not be used to get user details or safely determine if a user is logged
     * in, it's just for visual cues.
     *
     * This requires that the vue-cookie module be installed.
     *
     * @returns {(string|null)} the session id or null
     */
    getSession() {
      if (!this.$cookie) {
        return null
      }
 
      const cookie = this.$cookie.get('_u')
 
      if (cookie === '' || cookie === '""') {
        return null
      }
 
      return cookie
    },
 
    /**
     * Formats a telephone number for a link
     *
     * @param {string} telephoneNumber the telephone number to format
     * @returns {string} the formatted number
     */
    formatTelephoneNumberLink(telephoneNumber) {
      return telephoneLinkFormatter(telephoneNumber)
    },
  },
}
</script>
