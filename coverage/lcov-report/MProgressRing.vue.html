<template>
  <div class="m-circle-graph" :width="`${size}px`" :height="`${size}px`">
    <div class="m-graph-text">
      <slot />
    </div>
    <svg
      :width="`${size}px`"
      :height="`${size}px`"
      viewBox="0 0 36 36"
      class="m-donut"
    >
      <filter id="drop-shadow">
        <feDropShadow dx="0" dy="0.2" stdDeviation="0.4" flood-opacity="0.25" />
      </filter>
      <circle
        class="m-donut m-donut--full"
        :cx="centre"
        :cy="centre"
        :r="radius"
        :stroke-dasharray="`${percent} ${percentEmpty}`"
        :stroke-dashoffset="secondOffset"
      />
      <circle
        class="m-donut m-donut--empty"
        :cx="centre"
        :cy="centre"
        :r="radius"
        :stroke-dasharray="`${percentEmpty} ${percent}`"
        :stroke-dashoffset="firstOffset"
      />
    </svg>
  </div>
</template>
 
<script>
export default {
  name: 'MProgressRing',
  props: {
    percentFull: {
      type: Number,
      required: true,
    },
    size: {
      type: Number,
      required: true,
    },
  },
  data() {
    return {
      centre: 18,
      // A circle with this radius has a circumference of 100 (This makes calculations much easier).
      radius: 15.91549430918954,
      // By default dashes start at 90 degrees, this offest makes them start from the top of the circle.
      firstOffset: 25,
    }
  },
  computed: {
    percent() {
      return Math.min(Math.max(this.percentFull, 0), 100)
    },
    // At what point clockwise round the circle the grey line should start.
    secondOffset() {
      return this.firstOffset + this.percent
    },
    percentEmpty() {
      return 100 - this.percent
    },
  },
}
</script>
