<template>
  <form
    data-role="header-search-form"
    action="/search/"
    autocomplete="off"
    method="GET"
    class="m-search-form"
    :class="{ 'search-form--drawer': formType === 'drawer' }"
    @submit="submitSearch"
  >
    <fieldset>
      <input
        ref="searchInput"
        data-role="header-search-input"
        class="m-search-form__input"
        type="search"
        name="q"
        placeholder="Search Bought By Many"
        data-search-context=""
      />
      <m-search-button
        class="btn"
        :data-role="'search-submit'"
        :button-style="'greencircle'"
        :button-type="'submit'"
        :form-type="formType"
      ></m-search-button>
    </fieldset>
  </form>
</template>
 
<script>
import MSearchButton from './MSearchButton.vue'
 
export default {
  name: 'MSearchForm',
  components: {
    MSearchButton,
  },
  props: {
    formType: {
      type: String,
      required: false,
      default: null,
    },
    searchText: {
      type: String,
      required: false,
      default: 'Search',
    },
  },
  data() {
    return {}
  },
  mounted() {
    // TODO: Does this need to be in a nexttick?
    this.$refs.searchInput.focus()
  },
  methods: {
    submitSearch(event) {
      const searchString = event.target.q.value
      if (searchString === '') {
        this.$emit('searchFormHide')
        event.preventDefault()
      }
    },
  },
}
</script>
