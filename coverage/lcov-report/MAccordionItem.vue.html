<template>
  <div class="m-accordion-item">
    <button
      class="m-accordion-item--chevron"
      :class="{ 'chevron--close': isOpened }"
      @click="openItem()"
    ></button>
 
    <div class="m-accordion-item--title" @click="openItem()">
      {{ title }}
    </div>
 
    <template
      v-if="
        Array.isArray(subtitle) &&
          Array.isArray(subtitleLink) &&
          subtitle.length
      "
    >
      <nuxt-link
        v-for="(item, ix) in subtitle"
        :key="`link-${ix}`"
        :to="subtitleLink[ix]"
        class="m-accordion-item--subtitle"
      >
        {{ ix === 0 ? subtitle[ix] : ` | ${subtitle[ix]}` }}
      </nuxt-link>
    </template>
 
    <template
      v-if="
        typeof subtitle === 'string' &&
          typeof subtitleLink === 'string' &&
          subtitle
      "
    >
      <nuxt-link :to="subtitleLink" class="m-accordion-item--subtitle">
        {{ subtitle }}
      </nuxt-link>
    </template>
 
    <div v-if="posted" class="m-accordion-item--posted">
      Posted on: {{ posted }}
    </div>
 
    <div class="m-accordion-item--intro-text">
      {{ introText }}
    </div>
    <collapse-transition>
      <section v-if="isOpened" class="m-accordion-item--items item">
        <m-tag v-if="isNew" tag-style="color-sailor">New</m-tag>
 
        <div class="item-inner--content">
          <slot />
        </div>
 
        <div v-if="buttons.length" class="item-inner--links">
          <p>
            <nuxt-link
              v-for="(button, index) in buttons"
              :key="index"
              :to="button.link"
            >
              {{ button.text }}
            </nuxt-link>
          </p>
        </div>
      </section>
    </collapse-transition>
  </div>
</template>
 
<script>
import MTag from './MTag'
import { CollapseTransition } from 'vue2-transitions'
const re = /[./-]/
 
export default {
  name: 'MAccordionItem',
  components: {
    MTag,
    CollapseTransition,
  },
  props: {
    title: {
      type: String,
      required: true,
    },
    subtitle: {
      type: [String, Array],
      required: false,
      default() {
        return null
      },
    },
    subtitleLink: {
      type: [String, Array],
      required: false,
      default() {
        return ''
      },
    },
    posted: {
      type: String,
      required: false,
      default() {
        return null
      },
      validator(value) {
        // (D)D.(M)M.(YY)YY || (D)D-(M)M-(YY)YY || (D)D/(M)M/(YY)YY
        return /\d\d?[-./]\d\d?[-./]\d{2,4}/.test(value)
      },
    },
    introText: {
      type: String,
      required: false,
      default() {
        return null
      },
    },
    content: {
      type: String,
      required: false,
      default() {
        return null
      },
    },
    buttons: {
      type: Array,
      required: false,
      default() {
        return []
      },
      validator(btnArr) {
        return btnArr.every(btn => btn.text && btn.link)
      },
    },
    open: {
      type: Boolean,
      default: null,
    },
  },
  data() {
    return {
      daysLimitForNewLabel: 14,
      localOpen: false,
    }
  },
  computed: {
    isNew() {
      if (this.posted) {
        const dateArr = this.posted.split(re).reverse()
        const postedInDays = new Date(dateArr) / 86400000
        const nowInDays = new Date() / 86400000
        const isNew = nowInDays - postedInDays < this.daysLimitForNewLabel
        return isNew
      }
      return false
    },
    isOpened() {
      // If prop hasn't been set use local state
      return this.open === null ? this.localOpen : this.open
    },
  },
  methods: {
    openItem() {
      // Must maintain own state as well as talking to parent.
      this.localOpen = !this.localOpen
      this.$emit('requestOpen', this.open ? false : true)
    },
  },
}
</script>
