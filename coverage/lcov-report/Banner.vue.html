    <div v-if="heading || text || primaryButtonText" class="banner__content">

            v-bind:btnLink="primaryButtonLink"
            v-bind:target="primaryButtonTarget"
            v-bind:data-qa-locator="primaryButtonDataLocator"
            modifier="btn--primary btn--small btn--text-cta"
          >
            {{ primaryButtonText }}
          </site-button>
 
          <!-- Secondary Button with HR -->
          <div
            v-if="secondaryButtonText && secondaryButtonLink"
            class="hidden-xxs"
          >
            <hr class="hr hr--short" />
            <site-button
              v-bind:btn-link="secondaryButtonLink"
              v-bind:target="secondaryButtonTarget"
              v-bind:data-qa-locator="secondaryButtonDataLocator"
              modifier="btn--chevron-right btn--micro"
            >
              {{ secondaryButtonText }}
            </site-button>
          </div>
        </div>
      </div>
    </div>
 
    <div class="banner__media banner-media">
      <!-- Banner background image (responsive images set in methods) -->
      <div v-bind:style="banner" class="banner-media__image" />
 
      <!-- Overlay images -->
      <div class="banner-media__overlay">
        <!-- Limited to 4 items, as more than 4 will look bad and
        flow beyond the width of the banner on mobile views -->
        <img
          v-for="(overlay, index) in overlays.slice(0, 4)"
          v-bind:key="index"
          v-bind:src="overlay.source"
          v-bind:alt="overlay.altText"
          loading="lazy"
        />
        <div v-if="textBox" class="banner-textbox">
          {{ textBox }}
        </div>
      </div>
 
      <!-- Show a quote and/or image -->
      <div v-if="quote" class="banner-quote hidden-xxs hidden-xs">
        <div class="container">
          <blockquote>&ldquo;{{ quote }}&rdquo;</blockquote>
          <img
            v-if="quoteImage"
            v-bind:src="quoteImage"
            v-bind:alt="quoteAltText"
            loading="lazy"
          />
        </div>
      </div>
    </div>
  </div>

<script>
import HeadingOne from './HeadingOne.vue'
import SiteButton from './Button.vue'
import debounce from 'lodash.debounce'
 
export default {
  name: 'Banner',
  components: {
    HeadingOne,
    SiteButton,
  },
  props: {
    wideBannerImage: {
      type: String,
      required: false,
    },
    regularBannerImage: {
      type: String,
      required: true,
    },
    mobileBannerImage: {
      type: String,
      required: false,
    },
    heading: String,
    text: String,
    textBox: String,
    primaryButtonText: String,
    primaryButtonLink: String,
    primaryButtonTarget: String,
    primaryButtonDataLocator: {
      type: String,
      default: 'header-primary-button',
    },
    secondaryButtonText: String,
    secondaryButtonLink: String,
    secondaryButtonTarget: String,
    quote: String,
    quoteImage: String,
    quoteAltText: String,
    secondaryButtonDataLocator: {
      type: String,
      default: 'header-secondary-button',
    },
    overlays: {
      type: Array,
      required: false,
      default() {
        return []
      },
    },
  },
  data() {
    return { banner: '' }
  },
  mounted() {
    this.resizeBanner()
    window.addEventListener('resize', this.resizeBanner)
  },
  methods: {
    // Set regularBannerImage as default banner with added mobile & wide options
    resizeBanner: debounce(function() {
      if (window.innerWidth >= 1920 && this.wideBannerImage) {
        this.banner = `background-image: url('${this.wideBannerImage}');`
      } else if (window.innerWidth < 1024 && this.mobileBannerImage) {
        this.banner = `background-image: url('${this.mobileBannerImage}');`
      } else {
        this.banner = `background-image: url('${this.regularBannerImage}');`
      }
    }, 200),
  },
}
</script>
