<template>
  <div class="person-profile team-list__person">
    <div v-if="photo && displayPhoto">
      <img
        v-bind:src="photo"
        class="person-profile__img img-responsive"
        v-bind:alt="`Headshot of ${name}`"
        loading="lazy"
      />
    </div>
    <div>
      <div class="person-profile__name">
        {{ name }}
      </div>
      <div v-if="title && displayTitle" class="person-profile__title">
        {{ title }}
      </div>
      <a
        v-if="url && displayUrl"
        v-bind:href="url"
        class="btn btn--external-link"
        target="_blank"
        rel="noreferrer"
      >
        LinkedIn
      </a>
    </div>
  </div>
</template>
 
<script>
export default {
  name: 'team-member',
  props: {
    title: String,
    url: String,
    photo: String,
    type: {
      type: String,
      default: () => 'full',
    },
    name: {
      type: String,
      required: true,
    },
  },
 
  computed: {
    displayPhoto() {
      if (this.type === 'full' || this.type === 'noUrl') {
        return true
      }
      return false
    },
    displayTitle() {
      if (this.type !== 'nameOnly') {
        return true
      }
      return false
    },
    displayUrl() {
      if (this.type !== 'noUrl' && this.type !== 'nameOnly') {
        return true
      }
      return false
    },
  },
}
</script>
