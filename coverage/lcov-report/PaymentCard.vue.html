<template>
  <div
    class="payment-card"
    v-bind:class="[
      { 'payment-card--expired': expired },
      `payment-card--${brand}`,
      `payment-card--${fundingType}`,
    ]"
  >
    <div class="payment-card__inner">
      <div class="payment-card__elements">
        <span class="payment-card__chip"></span>
        <span v-if="brand || fundingType" class="payment-card__description">
          {{ brand }} {{ fundingType }}
        </span>
        <span v-if="expired" class="payment-card__warning">
          Card expired
        </span>
        <div class="payment-card__number">
          <span>* * * *</span>
          <span>* * * *</span>
          <span>* * * *</span>
          <span>{{ formattedTruncattedPan }}</span>
        </div>
        <span class="payment-card__expiry">
          Expiry date
          {{ expiry }}
        </span>
        <span class="payment-card__cardholder">
          {{ cardholder }}
        </span>
      </div>
    </div>
  </div>
</template>
 
<script>
export default {
  props: {
    brand: {
      type: String,
      default: '',
    },
    fundingType: {
      type: String,
      default: '',
    },
    truncatedPan: {
      type: String,
      default: '****',
    },
    cardholder: {
      type: String,
      default: null,
    },
    expired: {
      default: false,
      type: Boolean,
    },
    expiryMonth: {
      type: String,
      default: 'xx',
    },
    expiryYear: {
      type: String,
      default: 'xx',
    },
  },
  computed: {
    formattedTruncattedPan() {
      return this.truncatedPan.split('').join(' ')
    },
    expiry() {
      return `${this.formattedExpiryMonth}/${this.formattedExpiryYear}`
    },
    formattedExpiryMonth() {
      return this.expiryMonth.length === 1
        ? `0${this.expiryMonth}`
        : this.expiryMonth
    },
    formattedExpiryYear() {
      return this.expiryYear.length > 2
        ? this.expiryYear.substring(2)
        : this.expiryYear
    },
  },
}
</script>
