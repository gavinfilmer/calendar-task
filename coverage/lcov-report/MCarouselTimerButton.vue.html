<template>
  <div class="m-carousel-timer-button">
    <button class="btn" @click="handleClick">
      <div
        class="btn__chevron"
        :style="{
          height: `${radius / 2}px`,
          width: `${radius / 2}px`,
          borderTop: `${stroke}px solid ${strokeColor}`,
          borderRight: `${stroke}px solid ${strokeColor}`,
          left: `calc((50% - ${radius / 4}px) - ${stroke}px)`,
          top: `calc(50% - ${radius / 4}px)`,
        }"
      />
      <svg class="btn__progress" :height="2 * radius" :width="2 * radius">
        <circle
          :fill="fillColor"
          :stroke="strokeColor"
          :stroke-dasharray="circumference"
          :stroke-dashoffset="circumference"
          :style="{ strokeDashoffset }"
          :stroke-width="stroke"
          :r="normalizedRadius"
          :cx="radius"
          :cy="radius"
        />
      </svg>
    </button>
  </div>
</template>
 
<script>
export default {
  name: 'MCarouselTimerButton',
  props: {
    radius: {
      type: Number,
      required: false,
      default: () => 35,
    },
    progress: {
      type: Number,
      required: true,
    },
    stroke: {
      type: Number,
      required: false,
      default: () => 3,
    },
    mode: {
      type: String,
      required: false,
      default: () => 'dark',
      validator: val => ['light', 'dark', ''].includes(val),
    },
  },
  data() {
    const normalizedRadius = this.radius - this.stroke //* 2
    const circumference = normalizedRadius * 2 * Math.PI
 
    return {
      normalizedRadius,
      circumference,
    }
  },
  computed: {
    buttonStyle() {
      return [this.modifier]
    },
    strokeDashoffset() {
      return this.circumference - (this.progress / 100) * this.circumference
    },
    strokeColor() {
      return this.mode === 'light' ? 'rgba(255,255,255,1)' : 'rgba(0,0,0,1)'
    },
    fillColor() {
      return this.mode === 'light' ? 'rgba(255,255,255,.5)' : 'rgba(0,0,0,.5)'
    },
  },
  methods: {
    handleClick() {
      this.$emit('clicked')
    },
  },
}
</script>
