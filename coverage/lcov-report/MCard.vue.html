<template>
  <m-card-wrap
    class="m-prog-card-under"
    :class="{ grey: grey }"
    :raised="shadow"
    :chevron="chevron"
  >
    <div class="m-prog-card">
      <section class="content-container">
        <div class="arrow-container">
          <div
            v-if="this.$slots.detail"
            class="arrow-down"
            :class="{ 'arrow-up': cardOpen }"
            @click="openCard()"
          />
        </div>
        <div class="m-prog-card__top">
          <m-tag
            v-for="(tag, index) in tags"
            :key="index"
            :tag-style="tag.color"
            ><slot :name="tag.name" />
          </m-tag>
        </div>
        <slot />
        <div v-if="footer" class="footer font--micro">{{ footer }}</div>
      </section>
      <collapse-transition>
        <section v-if="cardOpen" class="card-detail">
          <slot name="detail" />
        </section>
      </collapse-transition>
    </div>
  </m-card-wrap>
</template>
 
<script>
import MTag from './MTag.vue'
import { CollapseTransition } from 'vue2-transitions'
import MCardWrap from './MCardWrap.vue'
 
export default {
  name: 'MCard',
  components: {
    MTag,
    MCardWrap,
    CollapseTransition,
  },
  props: {
    tags: {
      // Array of objects with name/colour keys
      type: Array,
      default() {
        return []
      },
      // Check every tag has a text value
      validator: tags => tags.every(tag => Object.keys(tag).includes('text')),
    },
    shadow: {
      type: Boolean,
      default: true,
    },
    grey: {
      type: Boolean,
      default: false,
    },
    chevron: {
      type: String,
      default: '',
      validator: value => ['', 'chevron-left', 'chevron-right'].includes(value),
    },
    footer: {
      type: String,
      default: '',
    },
    open: {
      type: Boolean,
      default: null,
    },
  },
  data() {
    return {
      localOpen: false,
    }
  },
  computed: {
    cardOpen() {
      // If prop hasn't been set use local state
      return this.open === null ? this.localOpen : this.open
    },
  },
  methods: {
    openCard() {
      // Must maintain own state as well as talking to parent.
      this.localOpen = !this.localOpen
      this.$emit('requestOpen', this.open ? false : true)
    },
  },
}
</script>
