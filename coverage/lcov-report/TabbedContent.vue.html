<template>
  <div v-if="tabs.length" class="tabbed-content">
    <ul class="tabs" v-bind:class="tabStyle">
      <li v-if="tabTitle" class="tabs__tab-title">
        {{ tabTitle }}
      </li>
      <li
        v-for="tab in tabs"
        v-bind:key="tab.id"
        v-bind:class="selectedTabId === tab.id ? 'tabs__tab--active' : ''"
        class="tabs__tab"
        role="button"
        v-on:click="selectedTabId = tab.id"
      >
        {{ tab.title }}
      </li>
    </ul>
 
    <div class="tabs__content">
      <component v-bind:is="selectedTab.component" v-bind="selectedTab.props" />
    </div>
  </div>
</template>
 
<script>
export default {
  name: 'tabbed-content',
  props: {
    tabTitle: {
      type: String,
      required: false,
      default: null,
    },
    tabs: {
      type: Array,
      required: false,
      default() {
        return []
      },
    },
    tabStyle: {
      type: String,
      required: true,
      default: null,
      validator: function(value) {
        // The value must match one of these strings
        return ['tabs--subnav', 'tabs--button-group'].includes(value)
      },
    },
    buttons: Boolean,
  },
  data() {
    return {
      selectedTabId: this.tabs.length ? this.tabs[0].id : null,
    }
  },
  computed: {
    selectedTab() {
      return this.tabs.find(({ id }) => this.selectedTabId === id)
    },
  },
}
</script>
